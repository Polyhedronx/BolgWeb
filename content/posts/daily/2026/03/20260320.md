## 1. 双指针接雨水

时间复杂度 $O(n)$，空间复杂度 $O(1)$。

```rust
impl Solution {
    pub fn trap(height: Vec<i32>) -> i32 {
        let mut left = 0;
        let mut right = height.len() - 1;
        let mut left_max = 0;
        let mut right_max = 0;
        let mut res = 0;

        while left < right {
            // 谁矮就动谁，因为能接的雨水量是根据弱侧来的
            if height[left] < height[right] {
                // 这里先判断弱侧是否最高
                // 维护一下 left_max 这个参数后续用于做差，得出单位宽度空间能接的雨水
                if height[left] > left_max {
                    left_max = height[left];
                } else {
                    res += left_max - height[left];
                }
                // 移动左指针一单位，进入下一次 while 循环
                left += 1;
            } else {
                // 类似的，只是左指针换成了右指针
                if height[right] > right_max {
                    right_max = height[right];
                } else {
                    res += right_max - height[right];
                }
                right -= 1;
            }
        }

        res
    }
}
```

## 2. 三数之和

```rust
impl Solution {
    pub fn three_sum(mut nums: Vec<i32>) -> Vec<Vec<i32>> {
        // 这里用 sort_unstable 而不是 sort，因为不需要保持相等元素的顺序（可以理解为不需要索引）
        // 这样速度会更快
        nums.sort_unstable();
        let len = nums.len();
        let mut res = Vec::new();
        
        for i in 0..len {
            // 剪枝1：排序后第一个数就是正数，那之后的数肯定也是正数，无法满足三数之和等于0
            if nums[i] > 0 {
                break;
            }
            // 剪枝2：与前一个数相同的情况不需要再计算一遍
            if i > 0 && nums[i] == nums[i - 1] {
                continue;
            }
            // 从当前数右侧开始放置左右指针
            let mut left = i + 1;
            let mut right = nums.len() - 1;
            // 这里是左右指针所指向的两数之和必须等于的目标数，即当前数的倒数
            let target = -nums[i];
            while left < right {
                let sum = nums[left] + nums[right];
                // Rust 特色的 match 语法，速度和可读性都优于 if-else
                match sum.cmp(&target) {
                    std::cmp::Ordering::Less => left += 1,
                    std::cmp::Ordering::Greater => right -= 1,
                    std::cmp::Ordering::Equal => {
                        // 先存进去这一组解
                        res.push(vec![nums[i], nums[left], nums[right]]);
                        // 通过两个 while 循环跳过左右侧相同的数
                        while left < right && nums[left] == nums[left + 1] {
                            left += 1;
                        }
                        while left< right && nums[right] == nums[right - 1] {
                            right -= 1;
                        }
                        // 同时移动左右指针寻找有无其它解
                        left += 1;
                        right -= 1;
                    }
                }
            }
        }
        
        res
    }
}
```

## 3. 盛最多水的容器

```rust
impl Solution {
    pub fn max_area(height: Vec<i32>) -> i32 {
        let mut left = 0;
        let mut right = height.len() - 1;
        let mut max = 0;

        while left < right {
            let w = (right - left) as i32;
            let h = height[left].min(height[right]);
            max = max.max(w * h);

            if height[left] < height[right] {
                left += 1;
            } else {
                right -= 1;
            }
        }

        max        
    }
}
```
