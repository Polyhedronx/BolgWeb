## 1. 单调队列：解决滑动窗口最值

```rust
use std::collections::VecDeque;

impl Solution {
    pub fn max_sliding_window(nums: Vec<i32>, k: i32) -> Vec<i32> {
        let len = nums.len();
        let k = k as usize;
        let mut res = Vec::with_capacity(len - k + 1);

        // 单调队列储存下标，不是值
        let mut deque = VecDeque::new();

        if nums.is_empty() || k <= 0 {
            return res;
        }

        // 维护队列单调递减，队列原有元素比新元素小的都出队，然后加进去
        for i in 0..len {
            while let Some(&tail) = deque.back() {
                // 遇到相等元素也要把原本的元素弹出，否则会残留多余元素
                if nums[i] >= nums[tail] {
                    deque.pop_back();
                } else {
                    break;
                }
            }
            deque.push_back(i);

            // 移除超出左边界的元素
            while let Some(&head) = deque.front() {
                // 这里最开始写的是 head <= i - k，但是 usize 类型不支持负号
                // 开始 i < k 的时候会溢出
                if head + k <= i{
                    deque.pop_front();
                } else {
                    break;
                }
            }

            // 输出当前队列最左边的元素即为当前窗口最大值
            if i >= k - 1 {
                if let Some(&head) = deque.front() {
                    res.push(nums[head]);
                }
            }
        }

        res

    }
}
```

## 2. [76. 最小覆盖子串](https://leetcode.cn/problems/minimum-window-substring/)

右指针 `right` 从 0 遍历到 `s_len-1`，共执行 n 次；

左指针 `left` 从 0 开始，最多也只会遍历到 `s_len-1`（每个字符最多被左指针访问 1 次）；

因此，**总操作次数是 n + n = 2n**。所以时间复杂度为 $O(n)$

需要的额外空间也是常数级别的，所以空间复杂度为 $O(1)$

```rust
impl Solution {
    pub fn min_window(s: String, t: String) -> String {
        let (s, t) = (s.as_bytes(), t.as_bytes());
        let (s_len, t_len) = (s.len(), t.len());
        let mut cnt = [0i32; 128];
        let mut diff = 0;

        // 剪枝
        if s_len < t_len {
            return String::new();
        }

        // 初始化词频数组和差异数
        for c in 0..t_len {
            let idx = t[c] as usize;
            if cnt[idx] == 0 {
                diff += 1;
            }
            // 注意初始化是减一，等会右侧进来要加一
            cnt[idx] -= 1;
        }

        // 滑动窗口
        let mut min_len = usize::MAX;
        let mut min_left = 0;
        let mut left = 0;
        for right in 0..s_len {
            // 右指针右移
            let right_char = s[right] as usize;
            cnt[right_char] += 1;
            if cnt[right_char] == 0 {
                diff -= 1;
            }
            // 当 diff 为 0 时开始缩小窗口左侧
            while diff == 0 {
                let left_char = s[left] as usize;
                let cur_len = right - left + 1;
                // 更新最小窗口长度和最小长度时的左指针
                if cur_len < min_len {
                    min_len = cur_len;
                    min_left = left;
                }
                // 移除左侧字符计数
                cnt[left_char] -= 1;
                // 移除后计数小于 0 说明 diff 增加，停止下一轮循环
                if cnt[left_char] < 0 {
                    diff += 1;
                }
                // 继续缩小窗口搜索更小的字串
                left += 1;
            }
        }

        // 最终处理
        if min_len == usize::MAX {
            String::new()
        } else {
            String::from_utf8(s[min_left..min_left + min_len].to_vec()).unwrap()
        }
    }
}
```
