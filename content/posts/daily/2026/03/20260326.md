## 1. [41. 缺失的第一个正数](https://leetcode.cn/problems/first-missing-positive/)

```rust
impl Solution {
    pub fn first_missing_positive(mut nums: Vec<i32>) -> i32 {
        // [1,3,5,7,6] -> swap(3,5) 将3放置到索引2的位置上
        // [1,5,3,7,6] -> 排序完再检查数组下标为i的位置上的数是不是i+1，发现i=1时不是，故答案是i+1=2
        // 再试几个：
        // [2,1,4,3,6] -> [1,2,4,3,6] -> [1,2,3,4,6] -> 6-1 超出数组下标范围答案就是数组最大下标i+1=5
        // [-1,-3,1,4] -> [1,-3,-1,4] -> 索引1上的数应该是2，不是的话则答案为2
        // 所以总共遍历数组两次，一次依靠 swap 排序，一次判断位置正确性
        // 注意：负数以及大于数组最大下标+1的数不参与 swap
        let len = nums.len();
        let mut i = 0;

        // 第一轮循环，归位所有正整数
        while i < len {
            let x = nums[i];
            if x > 0 && x <= len as i32 && x != nums[(x - 1) as usize] {
                let target =  (x - 1) as usize;
                nums.swap(i, target);
            } else {
                i += 1;
            }
        }

        // 第二轮循环，检查是否在对应位置上
        for (i, &num) in nums.iter().enumerate() {
            let i = i as i32;
            if num != i + 1 {
                return i + 1;
            }
        }

        // 否则返回最大的可能的正整数
        len as i32 + 1
    }
}
```

## 2. [73. 矩阵置零](https://leetcode.cn/problems/set-matrix-zeroes/)

```rust
impl Solution {
    pub fn set_zeroes(matrix: &mut Vec<Vec<i32>>) {
        // 一些基础矩阵参数
        let rows = matrix.len();
        let cols = matrix[0].len();
        let mut is_row1_0 = false;
        let mut is_col1_0 = false;

        // 记录一行和一列有没有 0
        for j in 0..cols {
            if matrix[0][j] == 0 {
                is_row1_0 = true;
                break;
            }
        }
        for i in 0..rows {
            if matrix[i][0] == 0 {
                is_col1_0 = true;
                break;
            }
        }

        // 矩阵遍历1：用第一行和第一列来当含有 0 的行和列的标记
        for i in 1..rows {
            for j in 1..cols {
                if matrix[i][j] == 0{
                    matrix[0][j] = 0;
                    matrix[i][0] = 0;
                }
            }
        }

        // 矩阵遍历2：对第一行有标记的行、第一列有标记的列进行 0 的替换
        for i in 1..rows {
            for j in 1..cols {
                if (matrix[0][j] == 0 || matrix[i][0] == 0) {
                    matrix[i][j] = 0;
                }
            }
        }

        // 处理第一行和第一列本身
        if is_row1_0 {
            for j in 0..cols {
                matrix[0][j] = 0;
            }
        }
        if is_col1_0 {
            for i in 0..rows {
                matrix[i][0] = 0;
            }
        }
    }
}
```

## 3. [54. 螺旋矩阵](https://leetcode.cn/problems/spiral-matrix/)

```rust
impl Solution {
    pub fn spiral_order(matrix: Vec<Vec<i32>>) -> Vec<i32> {
        let mut res = Vec::new();
        if matrix.is_empty() {
            return res;
        }

        // 开区间设计：[top, bottom)、[left, right)，永远不会出现 usize 下标溢出
        let mut top = 0;
        let mut bottom = matrix.len();
        let mut left = 0;
        let mut right = matrix[0].len();

        while top < bottom && left < right {
            // 遍历左到右
            for j in left..right {
                res.push(matrix[top][j]);
            }
            top += 1;

            // 遍历上到下
            for i in top..bottom {
                res.push(matrix[i][right-1]);
            }
            right -= 1;

            // 提前终止，避免单行/单列重复遍历
            if top >= bottom || left >= right {
                break;
            }

            // 遍历右到左
            for j in (left..right).rev() {
                res.push(matrix[bottom-1][j]);
            }
            bottom -= 1;

            // 遍历下到上
            for i in (top..bottom).rev() {
                res.push(matrix[i][left]);
            }
            left += 1;
        }

        res
    }
}
```
