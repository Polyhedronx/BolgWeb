## 1. [25. K 个一组翻转链表](https://leetcode.cn/problems/reverse-nodes-in-k-group/)

```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* reverseKGroup(ListNode* head, int k) {
        // 获取链表长度
        int len = 0;
        ListNode* pin = head;
        while (pin) {
            len += 1;
            pin = pin->next;
        }
        // k个一组循环，可以在外面初始化 curr 和 prev 节点
        // 因为内部循环时会不断覆盖不需要重复初始化
        ListNode dummy(0, head);
        ListNode* p0 = &dummy;
        ListNode* curr = head;
        ListNode* prev = nullptr;
        while (len >= k) {
            len -= k;
            for (int i = 0; i < k; ++i) {
                // 反转链表
                ListNode* next = curr->next;
                curr->next = prev;
                prev = curr;
                curr = next;
            }
            // 由于要移动（形式上的）哑节点，这里先记录应该要移动到的位置
            ListNode* temp = p0->next;
            // 将反转后的子链表尾部接到下一个待反转的子链表头部
            p0->next->next = curr;
            // 将哑节点接到反转后的子链表头部，这样就能重复上述循环步骤了
            p0->next = prev;
            // 当时这里要新建新的哑节点，很烦，所以选择移动哑节点的解法，但是这就需要先记录
            // 这里返回去记录
            p0 = temp;
        }
        return dummy.next;
    }
};
```
