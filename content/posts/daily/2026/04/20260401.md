## 1. [148. 排序链表](https://leetcode.cn/problems/sort-list/)

写这道题的时候第一个自然的想法是使用数组，但是显然空间复杂度不满足。

然后想到了归并排序，最自然的想法当然是递归，但这样的话时间复杂度是 $O(n)$，也不满足常数级要求。

顺着递归的思路走下去，如果不使用递归，应该要实现 `1->2->4->2n` 这样的分组并合并的逻辑，将合并的步骤封装成一个 `merge()` 函数并不难，就是最基础的排序算法。而要实现分组，需要同时实现组成一组和与其他组分离的方法。

1. 组合：使用一个 `for(int size = 1; size < len; size *= 2)` 来进行控制。
2. 分离：创建一个 `cut()` 函数，接收起始节点和 `size`，返回下一个分组的头节点。
3. 归并：两两 `merge()`。
4. 移动：将归并好的头节点移动到前一个归并好的头节点上。

```c++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode() : val(0), next(nullptr) {}
 *     ListNode(int x) : val(x), next(nullptr) {}
 *     ListNode(int x, ListNode *next) : val(x), next(next) {}
 * };
 */
class Solution {
public:
    ListNode* sortList(ListNode* head) {
        // 剪枝
        if (!head || !head->next) return head;
        // 非递归归并排序
        // 获取链表长度
        int len = 0;
        ListNode* pin = head;
        while (pin) {
            len += 1;
            pin = pin->next;
        }
        // 头节点可能被改变-->创建哑节点
        ListNode dummy(0, head);
        ListNode* prev = &dummy;
        ListNode* curr = prev->next;
        // 从底层开始归并，最小size=1
        for (int size = 1; size < len; size *= 2) {
            prev = &dummy;
            curr = prev->next;
            while (curr) {
                ListNode* left = curr;
                // 裁断链表，分成一个个待归并的子链表
                ListNode* right = cut(left, size);
                ListNode* next = cut(right, size);
                // 将归并后的头节点连接到前一个被归并好的尾节点上
                // 对于第一次特殊情况的含义是连接到初始哑节点，便于最后返回整体的头节点
                prev->next = merge(left, right);
                // 移动 prev 到该次被归并的链表尾节点
                while (prev->next) {
                    prev = prev->next;
                }
                // 移动curr 到下一次循环需要被归并的链表头节点
                curr = next;
            }
        }
        return dummy.next;
    }

private:
    ListNode* merge(ListNode* l1, ListNode* l2) {
        ListNode dummy;
        ListNode* curr = &dummy;
        while (l1 && l2) {
            if (l1->val < l2->val) {
                curr->next = l1;
                l1 = l1->next;
            } else {
                curr->next = l2;
                l2 = l2->next;
            }
            curr = curr->next;
        }
        curr->next = l1 ? l1 : l2;
        return dummy.next;
    }

    ListNode* cut(ListNode* head, int size) {
        // 裁断就是指向 nullptr，但是这个函数返回值要是被裁断的节点原本指向的下一个节点
        while (--size && head) {
            head = head->next;
        }
        if (!head) return head;
        ListNode* rest = head->next;
        head->next = nullptr;
        return rest;
    }
};
```

## 2. [23. 合并 K 个升序链表](https://leetcode.cn/problems/merge-k-sorted-lists/)

这一题也是使用和上一题类似的思想，从底层开始逐级归并。这里有几个重点：

1. 分组思路类似，依旧是利用 `for(int interval = 1; interval < len; interval *= 2)` 这样的循环。
2. 步长 `interval` 和上题 `size` 的区别就是，这里的步长需要在下一层级的流程控制中发挥 **选择需要归并的链表** 的作用，因为是原地解法不开辟新的内存空间，需要把一级一级归并好的链表存储在某些原位上，这里举例：`[0, 1, 2, 3, 4] -> [0&1, 1, 2&3, 3, 4] -> [0&1&2&3, 1, 2&3, 3, 4] -> [0&1&2&3&4, 1, 2&3, 3, 4]`。
3. 注意剪枝！

```c++
class Solution {
public:
    ListNode* mergeKLists(vector<ListNode*>& lists) {
        // 从最底层开始归并
        int len = lists.size();
        if (len == 0)
            return nullptr;
        for (int interval = 1; interval < len; interval *= 2) {
            for (int i = 0; i + interval < len; i += 2 * interval) {
                lists[i] = merge(lists[i], lists[i + interval]);
            }
        }
        return lists[0];
    }

    ListNode* merge(ListNode* l1, ListNode* l2) {
        ListNode dummy;
        ListNode* curr = &dummy;
        while (l1 && l2) {
            if (l1->val < l2->val) {
                curr->next = l1;
                l1 = l1->next;
            } else {
                curr->next = l2;
                l2 = l2->next;
            }
            curr = curr->next;
        }
        curr->next = l1 ? l1 : l2;
        return dummy.next;
    }
};
```

## 3. 双向链表实现 LRU 缓存

这题核心有四条：

1. 实现一个双向链表。
2. 使用哈希表储存双向链表上的节点，使用 `key - value`。
3. 创建虚拟头尾节点，如果只创建一个虚拟头节点会非常复杂，创建虚拟尾节点能帮助快速删除链表尾部的真实节点。
4. 拆分函数，让 `addToHead` / `moveToHead` / `removeNode` 等功能在 `private` 块中实现，同时也能简化对外暴露的 `get` / `put` API 的复杂度。

注意 `private` 块中需要传递存放诸如 `size`、`capacity` 等参数，防止外部恶意访问。

```c++
class LRUCache {
struct Node {
    int key;
    int value;
    Node* prev;
    Node* next;
    Node(): key(0), value(0), prev(nullptr), next(nullptr) {}
    Node(int _key, int _value): key(_key), value(_value), prev(nullptr), next(nullptr) {}
};

public:
    LRUCache(int capacity) {
        head = new Node(0, 0);
        tail = new Node(0, 0);
        head->next = tail;
        tail->prev = head;
        size = 0;
        this->capacity = capacity;
    }

    // 折构函数释放内存
    ~LRUCache() {
        Node* cur = head;
        while (cur) {
            Node* temp = cur;
            cur = cur->next;
            delete temp;
        }
        cache.clear();
    }
    
    int get(int key) {
        if (cache.find(key) == cache.end())
            return -1;
        Node* node = cache[key];
        // 标记为最近使用
        moveToHead(node);
        return node->value;
    }
    
    void put(int key, int value) {
        // 1.缓存命中，此时只需要更新值就可以
        if (cache.find(key) != cache.end()) {
            Node* node = cache[key];
            node->value = value;
            moveToHead(node);
            return;
        }
        // 2.缓存未命中，先插入再判断容量，决定是否移除尾节点
        Node* newNode = new Node(key, value);
        cache[key] = newNode;
        addToHead(newNode);
        if (size > capacity) {
            Node* delNode = tail->prev;
            cache.erase(delNode->key);
            removeNode(delNode);
        }
    }

private:
    unordered_map<int, Node*> cache;
    Node* head;
    Node* tail;
    int size;
    int capacity;

    void addToHead(Node* node) {
        node->next = head->next;
        node->prev = head;
        head->next->prev = node;
        head->next = node;
        size++;
    }

    void moveToHead(Node* node) {
        // 先删除再添置到头部
        node->next->prev = node->prev;
        node->prev->next = node->next;
        // 这里复用 addToNode 函数
        node->next = head->next;
        node->prev = head;
        head->next->prev = node;
        head->next = node;
    }

    void removeNode(Node* node) {
        node->next->prev = node->prev;
        node->prev->next = node->next;
        delete node;
        size--;
    }
};

/**
 * Your LRUCache object will be instantiated and called as such:
 * LRUCache* obj = new LRUCache(capacity);
 * int param_1 = obj->get(key);
 * obj->put(key,value);
 */
```
